1.charAt/charCodeAt

str = "사자/호랑이/고양이/치타/코끼리/표범";
charAt(index) :index위치의 문자를 리턴
charCodeAt(index) : index위치의 unicode값을 리턴



2. indexOf/lastIndexOf

str = "가나다/사자/호랑이/고양이/치타/코끼리/표범/사자/가나다";
indexOf("문자열") :문자열의 위치를 검색
lastIndexOf("문자열") : 문자열의 위치를 검색- 뒤에서부터 시작



3. concat/trim

str = "가나다/사자/호랑이/고양이/치타/코끼리/표범/사자/가나다";
str2 = "사과/복숭아/자두/참외/수박/바나나"
concat() :문자열합치기
str4 = "     사과 바나나 복숭아 자두 참외 수박     "
trim() : 앞뒤 공백제거



4. substring/slice/substr

str = "가나다/사자/호랑이/고양이/치타/코끼리/표범/사자/가나다";
substring(start, end) : start~ end-1 까지의 글자를 추출
slice(start, end) : start~ end-1 까지의 글자를 추출
substr(start,length) start 부터 length갯수 를 추추ㄹ



5. split

str = "가나다/사자/호랑이/고양이/치타/코끼리/표범/사자/가나다";
split(구분자, 갯수) : 구분자로 문자를 구분해서 배열로 리턴 한다



5. replace

str = "가나다/사자/호랑이/고양이/치타/코끼리/표범/사자/가나다";
replace(old, new) : old의 값을 new 의 값으로 변경
replace(//g) /와 /사이에 정규표현식을 사용할 수 있다 , g : global
